STEP 1

A Plain Chain

from langchain_anthropic import ChatAnthropic
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser

prompt = ChatPromptTemplate.from_messages([
    ("system", "You are a precise technical writer. Answer in under three sentences."),
    ("human", "{question}"),
])
model  = ChatAnthropic(model="claude-sonnet-4-5", temperature=0)

chain = prompt | model | StrOutputParser()

print(chain.invoke({"question": "What is PagedAttention?"}))

Pure LCEL. Three Runnables, one pipe.

Add Retrieval

from langchain_chroma import Chroma
from langchain_openai import OpenAIEmbeddings
from langchain_core.runnables import RunnablePassthrough

store = Chroma(persist_directory="./chroma",
               embedding_function=OpenAIEmbeddings(model="text-embedding-3-small"))
retriever = store.as_retriever(search_kwargs={"k": 4})

def format_docs(docs):
    return "\n\n---\n\n".join(f"[{d.metadata['source']}]\n{d.page_content}" for d in docs)

rag_prompt = ChatPromptTemplate.from_template("""
Answer from the context. Cite sources as [filename].
<context>{context}</context>
Question: {question}
""")

rag = (
    {"context": retriever | format_docs, "question": RunnablePassthrough()}
    | rag_prompt
    | model
    | StrOutputParser()
)

Parallel {context, question} + sequential prompt → model → parser.

STEP 3

Add Tools

from langchain_core.tools import tool

@tool
def search_corpus(query: str) -> str:
    """Search the local knowledge corpus."""
    docs = _store.similarity_search(query, k=4)
    return "\n\n---\n\n".join(f"[{d.metadata['source']}]\n{d.page_content}" for d in docs)

@tool
def search_web(query: str) -> str:
    """Search the web. Use only when the corpus is insufficient."""
    return "\n\n".join(f"[{h['url']}]\n{h['content']}" for h in _tavily.invoke(query))

@tool
def delete_document(source: str) -> str:
    """Permanently delete a document. Irreversible — require human approval."""
    return f"Deleted {source}."

Destructive tool deliberately kept — drives the HITL step later.

py-tutorial / 05
STEP 4

LangGraph Agent

from typing import Annotated, TypedDict
from langchain_core.messages import SystemMessage, ToolMessage
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages

TOOLS = [search_corpus, search_web, delete_document]
tools_by_name = {t.name: t for t in TOOLS}
SYSTEM = SystemMessage(content="Prefer search_corpus; use search_web only when corpus is insufficient. ...")
model = ChatAnthropic(model="claude-sonnet-4-5", temperature=0).bind_tools(TOOLS)

class AgentState(TypedDict):
    messages: Annotated[list, add_messages]

def call_model(state):
    return {"messages": [model.invoke([SYSTEM, *state["messages"]])]}

def run_tools(state):
    out = []
    for call in state["messages"][-1].tool_calls:
        try: result = tools_by_name[call["name"]].invoke(call["args"])
        except Exception as e: result = f"ERROR: {e}"
        out.append(ToolMessage(content=str(result), tool_call_id=call["id"]))
    return {"messages": out}

def route(state):
    return "tools" if state["messages"][-1].tool_calls else END

g = StateGraph(AgentState)
g.add_node("agent", call_model); g.add_node("tools", run_tools)
g.add_edge(START, "agent")
g.add_conditional_edges("agent", route, ["tools", END])
g.add_edge("tools", "agent")
app = g.compile()

py-tutorial / 07
STEP 6

Human-in-the-Loop on Destructive Tools

from langgraph.types import interrupt, Command

DESTRUCTIVE = {"delete_document"}

def approve_destructive(state):
    pending = state["messages"][-1].tool_calls
    destructive = [c for c in pending if c["name"] in DESTRUCTIVE]
    if not destructive:
        return {}                                   # no-op, fall through
    decision = interrupt({"prompt": "Approve these destructive calls?",
                          "calls": destructive})
    if decision != "approve":
        return {"messages": [ToolMessage(content="User refused.",
                                         tool_call_id=c["id"]) for c in destructive]}
    return {}

g.add_node("approve", approve_destructive)
g.add_conditional_edges("agent", route, ["approve", END])
g.add_edge("approve", "tools")
# Resume after approval:
app.invoke(Command(resume="approve"), config)

State persists across the pause — even if the process dies waiting for approval.
