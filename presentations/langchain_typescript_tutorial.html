STEP 1

A Plain Chain

import { ChatAnthropic } from "@langchain/anthropic";
import { ChatPromptTemplate } from "@langchain/core/prompts";
import { StringOutputParser } from "@langchain/core/output_parsers";

const prompt = ChatPromptTemplate.fromMessages([
  ["system", "You are a precise technical writer. Answer in under three sentences."],
  ["human", "{question}"],
]);
const model = new ChatAnthropic({ model: "claude-sonnet-4-5", temperature: 0 });
const chain = prompt.pipe(model).pipe(new StringOutputParser());

console.log(await chain.invoke({ question: "What is PagedAttention?" }));

.pipe() is the TS equivalent of Python's | — JS can't overload operators.

STEP 4

LangGraph Agent

import { StateGraph, Annotation, END, START, messagesStateReducer } from "@langchain/langgraph";
import { SystemMessage, ToolMessage, type BaseMessage } from "@langchain/core/messages";

const SYSTEM = new SystemMessage("Prefer search_corpus; use search_web only if the corpus is insufficient. ...");
const toolsByName = Object.fromEntries(TOOLS.map(t => [t.name, t]));
const model = new ChatAnthropic({ model: "claude-sonnet-4-5", temperature: 0 }).bindTools(TOOLS);

const State = Annotation.Root({
  messages: Annotation<BaseMessage[]>({ reducer: messagesStateReducer, default: () => [] }),
});

async function callModel(state: typeof State.State) {
  return { messages: [await model.invoke([SYSTEM, ...state.messages])] };
}
async function runTools(state: typeof State.State) {
  const last = state.messages.at(-1) as any;
  const out: ToolMessage[] = [];
  for (const c of last.tool_calls ?? []) {
    try {
      const r = await toolsByName[c.name].invoke(c.args);
      out.push(new ToolMessage({ content: String(r), tool_call_id: c.id }));
    } catch (e) { out.push(new ToolMessage({ content: `ERROR: ${e}`, tool_call_id: c.id })); }
  }
  return { messages: out };
}
function route(state: typeof State.State) {
  return (state.messages.at(-1) as any).tool_calls?.length ? "tools" : END;
}

export const graph = new StateGraph(State)
  .addNode("agent", callModel).addNode("tools", runTools)
  .addEdge(START, "agent")
  .addConditionalEdges("agent", route, ["tools", END])
  .addEdge("tools", "agent");
export const app = graph.compile();

STEP 6

Human-in-the-Loop

import { interrupt, Command } from "@langchain/langgraph";

const DESTRUCTIVE = new Set(["delete_document"]);

async function approveDestructive(state: typeof State.State) {
  const last = state.messages.at(-1) as any;
  const destructive = (last.tool_calls ?? []).filter((c: any) => DESTRUCTIVE.has(c.name));
  if (destructive.length === 0) return {};
  const decision = interrupt({ prompt: "Approve destructive calls?", calls: destructive }) as string;
  if (decision !== "approve") {
    return { messages: destructive.map((c: any) =>
      new ToolMessage({ content: "User refused.", tool_call_id: c.id })) };
  }
  return {};
}

// Resume after approval:
await app.invoke(new Command({ resume: "approve" }), config);

Mirror of the Python HITL pattern — identical semantics, idiomatic TS.

Streaming in Next.js

// app/api/chat/route.ts
import { HumanMessage } from "@langchain/core/messages";
import { app } from "@/server/hitlAgent";

export async function POST(req: Request) {
  const { thread_id, messages } = await req.json();
  const config = { configurable: { thread_id } };
  const encoder = new TextEncoder();

  const stream = new ReadableStream({
    async start(controller) {
      for await (const [tok] of app.stream(
        { messages: messages.map((m: any) => new HumanMessage(m.content)) },
        { ...config, streamMode: "messages" },
      )) {
        controller.enqueue(encoder.encode(
          `data: ${JSON.stringify({ content: tok.content })}\n\n`));
      }
      controller.close();
    },
  });
  return new Response(stream, { headers: { "Content-Type": "text/event-stream" } });
}

TAKEAWAYS

Choose Your Language

Pick TypeScript when

  • Your backend is already Node / TS
  • You're deploying to edge runtimes (Cloudflare Workers, Vercel Edge, Deno Deploy)
  • You want one codebase shared with a Next/SvelteKit/Remix frontend
  • You prefer Zod to Pydantic

Pick Python when

  • Your stack is Python already
  • You want Deep Agents preset (deepagents is Python-only today)
  • You want early access to new LangGraph features (~2-6 weeks ahead of TS)
  • Data/ML ecosystem integrations matter

Neither is wrong. The two SDKs are 95% interchangeable at the mental-model level — migrating either way is mechanical.
